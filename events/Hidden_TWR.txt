# Error checking to make sure subordinate countries peace out with the target
country_event = {
	id = hiddentwrwar.205

	hidden = yes
	is_triggered_only = yes

	trigger = {
		has_war_with = FROM
	}

	immediate = {
		log = "[GetDateText]: [Root.GetName]: event hiddentwrwar.205"
		white_peace = FROM
	}
}
