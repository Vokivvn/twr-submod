### Clears the language flags of a state
#
# How to use:
#
#	state = {
#		clear_language = yes
#	}
#
# Used for dynamic state names
#
#
clear_language = {
	hidden_effect = {
		if={
			limit={has_state_flag=russian_language}
			clr_state_flag = russian_language
		}
		else_if = {
			limit={has_state_flag=german_language}
			clr_state_flag = german_language
		}
		else_if = {
			limit={has_state_flag=is_ss_puppet}
			clr_state_flag = is_ss_puppet
		}
	}
}
